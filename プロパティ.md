## プロパティ
public class Counter
{
    public Counter(int count)
    {
        SetCount(count);
    }
    // ここからプロパティ
    public int Count
    {
        get
        {
            return _count;
        }

        set
        {
            if(value >= 100)
        {
            _count = 100;
            return;
        }
        _count = value;
        }
    }
    // private setにするとクラス内だけで値を変更できる
    public int Count{get;set;}

    public void Call()
    {
        _count++;
    }
    public void SetCount(int value)
    {
        if(value >= 100)
        {
            _count = 100;
            return;
        }
        _count = value;

        public int GetCount()
        {
            return _count;
        }
    }
}
